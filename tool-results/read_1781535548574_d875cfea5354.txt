    75тЖТ    75тЖТfunction AnimatedCounter({ value, className, style }: { value: number; className?: string; style?: React.CSSProperties }) {
    76тЖТ    76тЖТ  const [displayValue, setDisplayValue] = useState(value)
    77тЖТ    77тЖТ  const rafRef = useRef<number | null>(null)
    78тЖТ    78тЖТ  const prevValueRef = useRef(value)
    79тЖТ    79тЖТ
    80тЖТ    80тЖТ  useEffect(() => {
    81тЖТ    81тЖТ    if (prevValueRef.current !== value) {
    82тЖТ    82тЖТ      const duration = 600
    83тЖТ    83тЖТ      const startTime = Date.now()
    84тЖТ    84тЖТ      const startValue = prevValueRef.current
    85тЖТ    85тЖТ
    86тЖТ    86тЖТ      const animate = () => {
    87тЖТ    87тЖТ        const elapsed = Date.now() - startTime
    88тЖТ    88тЖТ        const progress = Math.min(elapsed / duration, 1)
    89тЖТ    89тЖТ        // Ease out cubic
    90тЖТ    90тЖТ        const eased = 1 - Math.pow(1 - progress, 3)
    91тЖТ    91тЖТ        setDisplayValue(Math.round(startValue + (value - startValue) * eased))
    92тЖТ    92тЖТ
    93тЖТ    93тЖТ        if (progress < 1) {
    94тЖТ    94тЖТ          rafRef.current = requestAnimationFrame(animate)
    95тЖТ    95тЖТ        } else {
    96тЖТ    96тЖТ          setDisplayValue(value)
    97тЖТ    97тЖТ        }
    98тЖТ    98тЖТ      }
    99тЖТ    99тЖТ      rafRef.current = requestAnimationFrame(animate)
   100тЖТ   100тЖТ      prevValueRef.current = value
   101тЖТ   101тЖТ    }
   102тЖТ   102тЖТ
   103тЖТ   103тЖТ    return () => {
   104тЖТ   104тЖТ      if (rafRef.current) cancelAnimationFrame(rafRef.current)
   105тЖТ   105тЖТ    }
   106тЖТ   106тЖТ  }, [value])
   107тЖТ   107тЖТ
   108тЖТ   108тЖТ  return (
   109тЖТ   109тЖТ    <motion.span
   110тЖТ   110тЖТ      className={className}
   111тЖТ   111тЖТ      style={style}
   112тЖТ   112тЖТ      key={value}
   113тЖТ   113тЖТ      initial={{ scale: 1.25 }}
   114тЖТ   114тЖТ      animate={{ scale: 1 }}
   115тЖТ   115тЖТ      transition={{ duration: 0.3, type: 'spring', bounce: 0.5 }}
   116тЖТ   116тЖТ    >
   117тЖТ   117тЖТ      {displayValue}
   118тЖТ   118тЖТ    </motion.span>
   119тЖТ   119тЖТ  )
   120тЖТ   120тЖТ}
